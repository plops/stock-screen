Creating a Python Script Screener
The best way to build a Python screener is to combine yfinance for raw fundamental fetching with pandas for dataframe filtering.
1. Technical Architecture

    Ticker Source: Fetch a list of tickers (e.g., the S&P 500) using pandas.read_html from Wikipedia.
    Data Extraction: Loop through tickers using yfinance to pull the info dictionary and financial statements.
    Filtering Engine: Apply conditional Pandas masks to bucket companies into Quadrants 1 through 4.

2. Implementation Code
python

import pandas as pd
import yfinance as yf
import time

def get_sp500_tickers():
    # Fetch S&P 500 ticker list from Wikipedia
    url = 'https://wikipedia.org'
    table = pd.read_html(url)
    return table[0]['Symbol'].str.replace('.', '-', regex=False).tolist()

def screen_stock(ticker_symbol):
    try:
        ticker = yf.Ticker(ticker_symbol)
        info = ticker.info
        
        # Extract core framework metrics
        rev_growth = info.get('revenueGrowth', 0)     # Decimal form (0.10 = 10%)
        eps_growth = info.get('earningsGrowth', 0)    # Decimal form
        forward_pe = info.get('forwardPE', None)
        debt_ebitda = info.get('debtToEbitda', None)  # Directly provided by Yahoo
        
        return {
            'Ticker': ticker_symbol,
            'Rev_Growth': rev_growth,
            'EPS_Growth': eps_growth,
            'Forward_PE': forward_pe,
            'Debt_EBITDA': debt_ebitda
        }
    except Exception:
        return None

# Example execution for a small test batch
tickers = get_sp500_tickers()[:20] 
screened_data = []

for t in tickers:
    data = screen_stock(t)
    if data:
        screened_data.append(data)
    time.sleep(0.5) # Polite rate limiting

df = pd.DataFrame(screened_data).dropna()

# Quadrant 4 Categorization Example (High Interest / QT: Defensive)
# Target: Stable, low-debt, low-valuation
q4_defensive = df[
    (df['Forward_PE'] < 15) & 
    (df['Debt_EBITDA'] < 2.0) & 
    (df['Rev_Growth'] > 0)
]
print(q4_defensive)
